```{r, SETTINGS-knitr, include=FALSE}
## knitr settings used to build vignettes
library(OncoBayes2)
library(posterior)
library(RBesT)
library(dplyr)
library(tidyr)
library(knitr)
library(ggplot2)
ggplot2::theme_set(bayesplot::bayesplot_theme_get())
if (Sys.which("pngquant") != "") {
  # compress plots if pngquant is on the system
  knitr::knit_hooks$set(pngquant = knitr::hook_pngquant)
  knitr::opts_chunk$set(
    dev = "ragg_png",
    dpi = 72,
    fig.retina = 1.5,
    fig.width = 1.62*4,
    fig.height = 4,
    fig.align = "center",
    out.width = "100%",
    pngquant = "--speed=1 --quality=50"
    )
}
```
